pyproject.toml

  • Configuration file used by packaging tools
  • Sections, also called tables are represented using square brackets [section name]
  • At least 3 sections are required (or recommended) -> [build-system], [project], [tool]
  • Python Packaging User Guide
  • Setuptools User Guide

Code example

[build-system]
requires = ["setuptools >= 61.0", "setuptools-scm"]
build-backend = "setuptools.build_meta"

[project]
name = "bhtp"
dynamic = ["version"]
description = "Package to mnsge tools for trading using the BHTP Platform"
readme = "README.md"
authors = [{name = "PoivronJaune", email="poivronjaune@gmail.com"}]
license = {file = "LICENSE"}
classifiers = [
    "License :: OSI Approved :: MIT License",
    "Programming Language :: Python",
    "Programming Language :: Python :: 3.10"
]
keywords = ["trading", "OHLC", "Minute prices", "Stockmarket"]
dependencies = [
    "pandas",
    "requests"
]
requires-python = ">3.10"

[project.optional-dependencies]
build = ["build","twine"]
dev = ["black", "pytest"]

[tool.setuptools.dynamic]
version = {attr = "bhtp.__version__"}
